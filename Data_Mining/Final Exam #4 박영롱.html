library(ggplot2)

partyFreq = function(data, cluster)
{
  outPut = NULL
  for (i in 1:length(cluster))
  {
    partySum = c()
    g = sum(data[,colnames(data) == names(cluster[i])][which(data$party == '')])
    d = sum(data[,colnames(data) == names(cluster[i])][which(data$party == '')])
    f = sum(data[,colnames(data) == names(cluster[i])][which(data$party == '')])
    b = sum(data[,colnames(data) == names(cluster[i])][which(data$party == '')])
    j = sum(data[,colnames(data) == names(cluster[i])][which(data$party == '')])
    m = sum(data[,colnames(data) == names(cluster[i])][which(data$party == '')])
    partySum = c(g, d, f, b, j, m) 
    names(partySum) = c('', '', '', '', '', '')
    
    outPut = rbind(outPut, partySum)
    rownames(outPut)[i] = names(cluster[i])
  }
  print('      ')
  print(outPut)
  result = colSums(outPut)/length(cluster)
  result2 = transform(result)
  result2 = cbind(rownames(result2), result2)
  colnames(result2) = c('Party', 'Proposition')
  pie<- ggplot(result2, aes(x='', y=Proposition, fill=Party))+
    geom_bar(width = 1, stat = "identity") + coord_polar("y", start = 0)
  plot(pie)
  return(result)
}
